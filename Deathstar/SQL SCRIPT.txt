--**********************************DDL NAREDBE**********************************

CREATE TABLE SAVEZ(

    ID NUMBER(20) CONSTRAINT SAVEZ_PK PRIMARY KEY,
    NAZIV VARCHAR(60) NOT NULL,
    DATUM_FORMIRANJA DATE NOT NULL,
    SAVEZ_ID NUMBER(20) REFERENCES SAVEZ(ID)

);

CREATE TABLE IGRAC(

    ID NUMBER(20) CONSTRAINT IGRAC_PK PRIMARY KEY,
    KORISNICKO_IME VARCHAR(60) NOT NULL,
    DATUM_OTVARANJA_NALOGA DATE NOT NULL,
    LICNO_IME VARCHAR(60) NOT NULL,
    SREDNJE_SLOVO CHAR(1) NOT NULL,
    PREZIME VARCHAR(60) NOT NULL,
    EMAIL VARCHAR(60),
    POL CHAR(1) CONSTRAINT POL_CK CHECK(POL IN('M', 'Z')),
    DATUM_RODJENJA DATE,
    DRZAVA VARCHAR(60),
    TEKSTUALNI_OPIS VARCHAR(100),
    SLIKA VARCHAR(60),
    SAVEZ_ID NUMBER(20) REFERENCES SAVEZ(ID)

);

CREATE TABLE GALAKSIJA(

    ID NUMBER(20) CONSTRAINT GALAKSIJA_PK PRIMARY KEY,
    IME VARCHAR(60) NOT NULL,
    DOMINANTNA_RASA VARCHAR(60),
    PROCENJENI_BR_PLANETA NUMBER(20),
    PROCENJENI_BR_ZVEZDA NUMBER(20)

);

CREATE TABLE PLANETA(

    ID NUMBER(20) CONSTRAINT PLANETA_PK PRIMARY KEY,
    NAZIV VARCHAR(60) NOT NULL,
    BROJ_STANOVNIKA NUMBER(20),
    DOMINANTNA_RASA VARCHAR(60),
    DRUSTVENO_UREDJENJE VARCHAR(60),
    TRILIJUM NUMBER(20),
    BERILIJUM NUMBER(20),
    PLUTONIJUM NUMBER(20),
    X NUMBER(10) NOT NULL,
    Y NUMBER(10) NOT NULL,
    Z NUMBER(10) NOT NULL,
    MATICNA CHAR(2) CONSTRAINT MATICNA_CK CHECK (MATICNA IN('DA', 'NE')),
    IGRAC_ID NUMBER(20) REFERENCES IGRAC(ID),
    GALAKSIJA_ID NUMBER(20) REFERENCES GALAKSIJA(ID)
);

CREATE TABLE ZVEZDA(

    ID NUMBER(20) CONSTRAINT ZVEZDA_PK PRIMARY KEY,
    IME VARCHAR(60) NOT NULL,
    TIP VARCHAR(60)

);


CREATE TABLE SPEC_POJAVA(

    ID NUMBER(20) CONSTRAINT SPEC_POJAVA_PK PRIMARY KEY,
    NAZIV VARCHAR(60) NOT NULL,
    TIP VARCHAR(60),
    POT_OPASNOST CHAR(2) CONSTRAINT POT_OPASNOST_CK CHECK (POT_OPASNOST IN ('DA', 'NE'))

);

CREATE TABLE GRAD(

    ID NUMBER(20) CONSTRAINT GRAD_PK PRIMARY KEY,
    NAZIV VARCHAR(30) NOT NULL,
    GLAVNI CHAR(2) CONSTRAINT GLAVNI_CK CHECK (GLAVNI IN('DA', 'NE')),
    PLANETA_ID NUMBER(20) REFERENCES PLANETA(ID)

);

CREATE TABLE PRIRODNI_SATELIT(

    ID NUMBER(20) CONSTRAINT PRIRODNI_SATELIT_PK PRIMARY KEY,
    NAZIV VARCHAR(60) NOT NULL,
    PRECNIK NUMBER(20),
    NASEOBINE CHAR(2) CONSTRAINT NASEOBINE_CK CHECK (NASEOBINE = 'DA' OR NASEOBINE = 'NE'),
    UDALJENOST NUMBER(10),
    PLANETA_ID NUMBER(20) REFERENCES PLANETA(ID)
);


CREATE TABLE VOJNA_STANICA(

    ID NUMBER(20) CONSTRAINT VOJNA_STANICA_PK PRIMARY KEY,
    NAZIV VARCHAR(60) NOT NULL,
    BR_LJUDI NUMBER(20),
    VELICINA NUMBER(20),
    UDALJENOST NUMBER(20),
    SPISAK_ORUZJA VARCHAR(1000),
    PLANETA_ID NUMBER(20) REFERENCES PLANETA(ID)

);

CREATE TABLE CIVILNA_STANICA(

    ID NUMBER(20) CONSTRAINT CIVILNA_STANICA_PK PRIMARY KEY,
    NAZIV VARCHAR(60) NOT NULL,
    BR_LJUDI NUMBER(20),
    VELICINA NUMBER(20),
    UDALJENOST NUMBER(20),
    ZA_TRGOVINU CHAR(2) CONSTRAINT ZA_TRGOVINU_CK CHECK (ZA_TRGOVINU IN('DA', 'NE')),
    ZA_NAUCNA_ISTRAZIVANJA CHAR(2) CONSTRAINT ZA_NAUCNA_ISTRAZIVANJA_CK CHECK (ZA_NAUCNA_ISTRAZIVANJA IN('DA', 'NE')),
    PLANETA_ID NUMBER(20) REFERENCES PLANETA(ID)
);

CREATE TABLE BROD(
    
    JEDINSTVENI_BROJ NUMBER(20) CONSTRAINT BROD_PK PRIMARY KEY,
    NAZIV VARCHAR(60) NOT NULL,
    MAX_WARP NUMBER(20),
    BROJ_LASERSKIH_TOPOVA NUMBER(20),
    FOTONSKA_TORPEDA CHAR(2) CONSTRAINT FOTONSKA_TORPEDA_CK CHECK (FOTONSKA_TORPEDA IN('DA', 'NE')),
    TIP VARCHAR(60) CONSTRAINT TIP_CK CHECK (TIP IN ('RAZARAC', 'KRSTARICA', 'FREGATA')),
    ZASTITNA_OPLATA CHAR(2) CONSTRAINT ZASTITNA_OPLATA_CK CHECK (ZASTITNA_OPLATA IN('DA', 'NE')),
    NOSIVOST NUMBER(20),
    VRSTA_BRODA VARCHAR(60) CONSTRAINT VRSTA_BRODA_CK CHECK (VRSTA_BRODA IN('BORBENI', 'TRANSPORTNI')),
    IGRAC_ID NUMBER(20) REFERENCES IGRAC(ID),
    SAVEZ_ID NUMBER(20) REFERENCES SAVEZ(ID),
    PLANETA_ID NUMBER(20) REFERENCES PLANETA(ID)
);

CREATE TABLE KOLONIZACIJA(

    ID NUMBER(20) CONSTRAINT KOLONIZACIJA_PK PRIMARY KEY,
    DATUM DATE,
    IGRAC_ID NUMBER(20) REFERENCES IGRAC(ID),
    SAVEZ_ID NUMBER(20) REFERENCES SAVEZ(ID),
    PLANETA_ID NUMBER(20) REFERENCES PLANETA(ID)

);

CREATE TABLE OSVAJANJE(

    ID NUMBER(20) CONSTRAINT OSVAJANJE_PK PRIMARY KEY,
    DATUM DATE,
    PRETHODNI_VLASNIK_IGRAC VARCHAR(60),
    PRETHODNI_VLASNIK_SAVEZ VARCHAR(60),
    IGRAC_ID NUMBER(20) REFERENCES IGRAC(ID),
    SAVEZ_ID NUMBER(20) REFERENCES SAVEZ(ID),
    PLANETA_ID NUMBER(20) REFERENCES PLANETA(ID)

);

CREATE TABLE KVADRANT(
    
    ID NUMBER(20) CONSTRAINT KVADRANT_PK PRIMARY KEY, --ZBOG MAPIRANJA U NHIBERNATE DA BI SE IZBEGAO KOMPOZITNI KLJUC
    PROCENJENI_PRECNIK NUMBER(20),
    REDNI_BROJ NUMBER(20) CHECK(REDNI_BROJ BETWEEN 0 AND 6),
    GALAKSIJA_ID NUMBER(20) REFERENCES GALAKSIJA(ID)
    --CONSTRAINT KVADRANT_PK PRIMARY KEY (REDNI_BROJ, GALAKSIJA_ID)

);

CREATE TABLE ZVEZDA_PLANETA_BLIZINA(

    ID NUMBER(20) CONSTRAINT ZVEZDA_PLANETA_BLIZINA_PK PRIMARY KEY, --IZBEGAVANJE KOMPOZITNOG KLJUCA
    PLANETA_ID NUMBER(20) REFERENCES PLANETA(ID),
    ZVEZDA_ID NUMBER(20) REFERENCES ZVEZDA(ID)
    --CONSTRAINT ZVEZDA_PLANETA_BLIZINA_PK PRIMARY KEY (PLANETA_ID, ZVEZDA_ID)

);

CREATE TABLE PLANETA_POJAVA_BLIZINA(

    ID NUMBER(20) CONSTRAINT PLANETA_POJAVA_BLIZINA_PK PRIMARY KEY, --IZBEGAVANJE KOMPOZITNOG KLJUCA
    PLANETA_ID NUMBER(20) REFERENCES PLANETA(ID),
    SPEC_POJAVA_ID NUMBER(20) REFERENCES SPEC_POJAVA(ID),
    UDALJENOST NUMBER(20)
    --CONSTRAINT PLANETA_POJAVA_BLIZINA_PK PRIMARY KEY (PLANETA_ID, SPEC_POJAVA_ID)

);
